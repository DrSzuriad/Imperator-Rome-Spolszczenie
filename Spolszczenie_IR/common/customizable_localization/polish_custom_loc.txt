			has_city_status = no

		}
	}

	text = {
		localization_key = cloc_shrine_BIER
		trigger = {
			has_city_status = no
		}
	}
	
	text = {
		localization_key = cloc_temple_BIER
		trigger = {
			has_city_status = yes

		}
	}

		}
	}
	
	text = {

		}
